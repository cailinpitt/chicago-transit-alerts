# Push updated alert data to GitHub Pages.
#
# This is a backstop only — cta-insights's src/shared/webPushTrigger.js
# fires the same script directly whenever a detection script posts to
# Bluesky, so the typical incident path lands on the public dashboard
# within seconds. The cron exists to catch the rare cases the trigger
# doesn't cover (alerts the bot recorded but didn't post, failed trigger
# spawns).
#
# Logs go to cron/push-web-data-cron.log (gitignored).
*/15 * * * * PAGES_REPO=$HOME/Development/cta-alert-history CTA_INSIGHTS=$HOME/Development/cta-insights /bin/sh $HOME/Development/cta-insights/bin/push-web-data.sh >> $HOME/Development/cta-alert-history/cron/push-web-data-cron.log 2>&1
